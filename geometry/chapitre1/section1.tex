	\begin{definition}
	Soient $X$ un ensemble et $V$ un espace vectoriel sur un corp $\Set{K}$ (abélien). \newline
	On dit que $X$ est un espace affine de direction $V$ s'il existe une application de $X \times V \rightarrow X, (x,\vec{v}) \mapsto x + \vec{v}$ \newline
	telle que:
	\begin{enumerate}[(i)]
		\item	$x + \vec{0} = x, \forall x \in X$
		\item	$(x + \vec{u}) + \vec{v} = x + (\vec{u} + \vec{v}) , \forall x \in X, \forall \vec{u},\vec{v} \in V$
		\item	$\forall x \in X, \forall y \in X , \exists ! \vec{v} \in V , y = x + \vec{v}$
	\end{enumerate}
	$\dim_{\Set{K}} X := \dim_{\Set{K}} V$. En plus $\emptyset$ est un espace affine pour lequel $\dim$ n'est pas défini.
	\end{definition}
	\begin{remark}
		(i) et (ii) nous montre que $V$ agit sur $X$, (iii) nous montre que l'action est simplement transitive
	\end{remark}
	\begin{definition}
		(Alternative) \newline	
		$X$ est un espace affine dirigé par $V$ s'il existe une action simplement transitive de $V$ sur $X$
	\end{definition}
	\fbox{Notation: } Si $\vec{v} \in V$, on définit $\tau_{\vec{v}} : X \rightarrow X, x \mapsto x + \vec{v}$
